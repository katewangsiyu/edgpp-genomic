\subsection{Marginal and class-conditional coverage}

\begin{theorem}[T1 marginal coverage]
\label{thm:t1}
Under Assumptions~\ref{ass:A1}--\ref{ass:A2}, for test point $(X, Y) \sim P_{c^\star}$ independent of the calibration fold,
\[
1 - \alpha \leq P\bigl(Y \in \pset(X)\bigr) \leq 1 - \alpha + \frac{1}{n_{\min} + 1}.
\]
\end{theorem}

\begin{theorem}[T2 class-conditional coverage]
\label{thm:t2}
Under the conditions of Theorem~\ref{thm:t1}, for each $k \in \{0, 1\}$ with $n_k \geq \lceil 1/\alpha\rceil$,
\[
1 - \alpha \leq P\bigl(Y \in \pset(X) \mid Y = k\bigr) \leq 1 - \alpha + \frac{1}{n_k + 1}.
\]
\end{theorem}

\subsection{Bin-conditional (local) coverage --- T3 family}

\begin{theorem}[T3 bin-conditional coverage, exact]
\label{thm:t3}
Under Assumptions~\ref{ass:A1}--\ref{ass:A2} with A2 at the cell level, for each cell $(k, b)$ with $n_{k,b} \geq \lceil 1/\alpha \rceil$,
\[
1 - \alpha \leq P\bigl(Y \in \pset(X) \mid Y = k, b(X) = b\bigr) \leq 1 - \alpha + \frac{1}{n_{k,b} + 1}.
\]
\end{theorem}

\textit{Connection to Barber 2020 impossibility}: Theorem~\ref{thm:t3} is a coarsened (bin-conditional) guarantee, which Barber 2020 does not preclude; it complements rather than contradicts the pointwise impossibility. Proof in App.~\ref{app:proofs}: apply Theorem~\ref{thm:t2}'s argument within each $(k, b)$ cell using A2-cell to replace A2-class.

\begin{theorem}[T3' bin-conditional coverage, robust]
\label{thm:t3prime}
Under A1 alone (no A2-cell),
\[
P\bigl(Y \in \pset(X) \mid Y = k, b(X) = b\bigr) \geq 1 - \alpha - \bar\delta_{\mathrm{TV}}^{(k,b)},
\]
where $\bar\delta_{\mathrm{TV}}^{(k,b)}$ is the average within-cell chromosome total-variation distance.
\end{theorem}

\begin{theorem}[T3-loc feature-ball coverage]
\label{thm:t3loc}
If $\hat\sigma$ is $L$-Lipschitz and $A3_{\mathrm{loc}}$ (local score stationarity) holds, then for any $x_0 \in \X$ and $r < \Delta / (2L)$,
\[
\left| P\bigl(Y \in \pset(X) \mid X \in B_\ell(x_0, r), Y = k\bigr) - (1 - \alpha) \right| \leq \frac{1}{n_{\min}(x_0, r) + 1}.
\]
\end{theorem}

\begin{theorem}[T3.b $\hat\sigma$ perturbation]
\label{thm:t3b}
If $\|\hat\sigma - \sigma^\star\|_\infty / \inf_x \sigma^\star(x) \leq \eta$, then
\[
\left|\,\mathrm{cov}^{(\hat\sigma)} - \mathrm{cov}^{(\sigma^\star)}\,\right| \leq \frac{2\eta\,\bar\sigma}{\Delta} + \frac{1}{n_{\min} + 1}.
\]
\end{theorem}

\subsection{Chrom-shift robustness}

\begin{theorem}[T4 chrom-shift bound --- Barber 2023 Thm 2 applied]
\label{thm:t4}
Let $\tilde{w}_i$ be weights assigned to calibration-fold chroms. Then
\[
P\bigl(Y \in \pset(X)\bigr) \geq 1 - \alpha - \sum_i \tilde{w}_i \cdot d_{TV}\bigl(R(Z), R(Z^{(i)})\bigr),
\]
where $R(Z)$ is the residual vector and $R(Z^{(i)})$ is the swap-one residual vector.
\end{theorem}
