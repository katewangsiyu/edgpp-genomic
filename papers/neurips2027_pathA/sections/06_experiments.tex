\subsection{Setup}

\paragraph{Datasets.} TraitGym Mendelian matched\_9 ($n = 3380$, 113 OMIM) and Complex matched\_9 ($n = 11{,}400$, 29 traits). Chrom-LOO protocol (test chroms $\{17, 18, 19, 20, 21, 22, X\}$).

\paragraph{Features.} CADD \citep{kircher2014cadd}, Borzoi\_L2\_L2 \citep{linder2025borzoi}, GPN-MSA\_absLLR \citep{benegas2023gpnmsa}. Four feature sets: CADD, Borzoi, CADD+Borzoi, CADD+GPN-MSA+Borzoi.

\paragraph{Metrics.} AUPRC weighted by test-chrom size (TraitGym leaderboard primary); marginal coverage; class-conditional coverage; \emph{$\shat$-bin coverage gap}, the max-over-bins absolute deviation of per-bin coverage from target $1 - \alpha$. All results are at $\alpha = 0.10$, seed 42, $K = 5$ unless noted.

\paragraph{Three observations.}
\begin{enumerate}
    \item \textbf{HCCP holds across all three axes}: $\shat$-bin gap $\leq 0.04$ in all settings except the strong-shift M$\to$C direction (see below). Chrom-LOO Mendelian CADD+Borzoi (gap $0.198$) is the residual pain point; CADD+GPN-MSA+Borzoi corrects it to $0.077$.
    \item \textbf{Trait-LOO reaches the finite-sample floor.} Gaps of $0.002\text{--}0.004$ match the $1/(n_{k,b}+1)$ bound of T3; per-OMIM coverage std is $\leq 0.052$, within binomial noise at $n = 30$ OMIMs.
    \item \textbf{Cross-dataset shift reveals direction asymmetry.} Complex $\to$ Mendelian retains $\mathrm{AUPRC} = 0.665$ and $\shat$-bin gap $0.035$ — evidence that the Complex feature space \emph{contains} the Mendelian one. Mendelian $\to$ Complex collapses (AUPRC $0.205$). This is biologically interpretable: Mendelian monogenic signal is a narrow subspace of the polygenic Complex signal, not vice versa. T4's Barber 2023 Thm 2 proxy bound is tight on Mendelian $\to$ Complex CADD+Borzoi: observed coverage $0.738$ vs proxy $(1 - \alpha) - \hat d_{\text{TV}} = 0.742$ (1.4 std below).
\end{enumerate}

\subsection{Ablations}

\paragraph{$K$ sweep (T3 finite-sample rate validation).} Across $K \in \{3, 5, 10, 20\}$ on Complex CADD+Borzoi: empirical $\shat$-bin gap grows as $K$ grows (smaller $n_{k,b}$), but stays below the theoretical $1/(n_{k,b}+1)$ bound (App.~\ref{app:tables}). $K = 5\text{--}10$ is the sweet spot.

\paragraph{$\shat$ source ablation.} We compare four heads: (a) supervised GBM on OOF residuals (HCCP default), (b) DEGU-lite bootstrap-GBM ensemble ($M = 10$, $\shat$ = ensemble std), (c) DEGU-style full NLL head (deferred, App.~\ref{app:tables}), (d) no $\shat$ (homoscedastic, Day 10). On Complex, HCCP (a) achieves gap $0.020$; DEGU-lite (b) gives $0.103$ — \emph{$5\times$ worse}. The failure mode of (b) is interpretable: bin 0 (low ensemble std, i.e.\ high agreement) has coverage $0.570$. All 10 bootstrap students agree \emph{but are wrong together}; ensemble disagreement captures epistemic uncertainty but not the aleatoric residual structure our score requires.

\paragraph{Class-conditional on/off.} With class-cond on and no $\shat$-bin stratification, gap on Complex is $0.51$ (Day 10). With $\shat$-bin on but no class-cond, minority-class coverage collapses (Cov\,$|Y{=}1 = 0.54$). Both are needed; the product partition is not decomposable.

\paragraph{Benchmark audit byproduct.} CADD-alone LogReg on Mendelian reaches AUPRC $0.871$, exceeding TraitGym's published ``SOTA'' CADD+Borzoi LogReg of $0.752$ \citep{benegas2025traitgym}. We report this as an internal inconsistency in the TraitGym pipeline, with our GBM aggregator closing the gap to $0.900$.

\subsection{Honest limitations}

\begin{itemize}
    \item \textbf{Marginal-score TV proxy.} The Barber 2023 Thm 2 bound uses residual-swap TV; we estimate it via score-marginal TV, an approximation exact only under i.i.d.\ but underestimating for cross-dataset shift. On CADD+GPN-MSA+Borzoi M$\to$C the proxy is violated by $0.036$. Future work: direct residual-swap TV estimation (§\ref{sec:discussion}).
    \item \textbf{DEGU reimpl caveat.} DEGU-lite shares the same GBM backbone as HCCP rather than the original CNN; we chose this for like-for-like comparison. A full DEGU CNN reimpl (deferred to App.~\ref{app:tables}) would isolate the conformal-partition contribution from the $\shat$-head contribution more cleanly.
    \item \textbf{A3-loc (local stationarity) cannot be fully verified} at our sample sizes. Our cell-level KS audit (12.3\% rejection at $p < 0.05$, App.~\ref{app:data}) is the closest empirical analog. T3-loc is stated conditionally on A3-loc throughout.
\end{itemize}
